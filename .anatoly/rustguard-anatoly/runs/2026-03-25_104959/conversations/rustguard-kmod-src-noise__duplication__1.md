/// BLAKE2s-256 hash of concatenated chunks.
fn hash(chunks: &[&[u8]]) -> [u8; 32] {
    let ptrs: [*const u8; 8] = {
        let mut p = [core::ptr::null(); 8];
        for (i, c) in chunks.iter().enumerate().take(8) {
            p[i] = c.as_ptr();
        }
        p
    };
    let lens: [u32; 8] = {
        let mut l = [0u32; 8];
        for (i, c) in chunks.iter().enumerate().take(8) {
            l[i] = c.len() as u32;
        }
        l
    };
    let mut out = [0u8; 32];
    unsafe {
        wg_blake2s_hash(ptrs.as_ptr(), lens.as_ptr(), chunks.len() as u32, out.as_mut_ptr());
    }
    out
}

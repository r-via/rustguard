unsafe fn mmap_ring(
    fd: i32,
    off: &XdpRingOffset,
    ring_size: u32,
    pgoff: i64,
    desc_size: usize,
) -> io::Result<*mut u8> {
    let size = off.desc as usize + (ring_size as usize) * desc_size;
    let ptr = libc::mmap(
        std::ptr::null_mut(),
        size,
        libc::PROT_READ | libc::PROT_WRITE,
        libc::MAP_SHARED | libc::MAP_POPULATE,
        fd,
        pgoff,
    );
    if ptr == libc::MAP_FAILED {
        return Err(io::Error::last_os_error());
    }
    Ok(ptr as *mut u8)
}

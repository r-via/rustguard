fn rand_index() -> u32 {
    let mut buf = [0u8; 4];
    getrandom::getrandom(&mut buf).expect("rng");
    u32::from_le_bytes(buf)
}


fn base64_key(key: &[u8; 32]) -> String {
    use base64::prelude::*;
    BASE64_STANDARD.encode(key)
}

/// Set up AF_XDP: load BPF program, create XSK, register in XSKMAP.

        ifname: ifname.to_string(),
        queue_id: 0,
        frame_size: 4096,

## Key Generation

RustGuard provides built-in key generation utilities. Keys are standard WireGuard-compatible base64-encoded X25519 values.

```bash
# Generate a private key
rustguard genkey

# Derive the corresponding public key
rustguard genkey | rustguard pubkey
```

The output of `genkey` is a 44-character base64 string suitable for use as `PrivateKey` in a `wg.conf` file or for any internal use requiring an X25519 key pair.
